#include <iostream>
#include <vector>

int main() {
    int n;
    double l, s = 0.0;

    std::cout << "Ile: ";
    std::cin >> n;

    std::vector<double> lby;

    std::cout << "L:" << std::endl;

    for (int i = 0; i < n; ++i) {
        std::cin >> l;
        lby.push_back(l);
        s += l;
    }

    double sr = s / n;

    std::cout << "Åšr: " << sr << std::endl;

    return 0;
}
