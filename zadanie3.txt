#include <vector>

int main() {

    double l, s = 0.0;

    std::cout << "Ile: ";
    std::cin >> n;

    std::vector<double> l;

    std::cout << "L:" << std::endl;

    for (int i = 0; i < n; ++i) {

        lby.push_back(l);
        s += l;

}
